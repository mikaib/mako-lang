.add (x: i32, y: i32) f32
    $0 = ADD x y
    $1 = CAST $0 f32
    RET $1

.identity<T=i32> (x: T) T
    RET x

.identity<T=f32> (x: T) T
    RET x

.main () f32
    multiplier = VAR f32
    $0 = CONST f32 2.0
    STORE multiplier $0
    $1 = CONST i32 10
    $2 = CALL identity<T=i32> $1
    $3 = CONST f32 3.14
    $4 = CALL identity<T=f32> $3
    $5 = CALL add $2 $2
    $6 = LOAD multiplier
    $7 = MUL $4 $5
    $8 = MUL $7 $6
    RET $8
